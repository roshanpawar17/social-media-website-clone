CREATE TABLE users( id int(20) NOT null AUTO_INCREMENT PRIMARY KEY, username varchar(25) NOT null UNIQUE, firstname varchar(20) NOT null, lastname varchar(20) NOT null, fullname varchar(50) NOT null, email varchar(40) NOT null, password varchar(25) NOT null, datetime varchar(50) NOT null , defaultprofile varchar(100) NOT null DEFAULT 'default.png' ) CREATE table description( idbio int(20) not null AUTO_INCREMENT PRIMARY key, bio varchar(500), idabout int(20) not null, FOREIGN key(idabout) REFERENCES users(id) ) CREATE table posts( pid int(20) not null AUTO_INCREMENT PRIMARY key, location varchar(20), post varchar(20) not null, caption varchar(500), idpost int(20) NOT null, FOREIGN key(idpost) REFERENCES users(id) ); ); );
